library(lmtest)

# Fit FE model
fe_model <- lm(EVSales ~ TK*, data = data, model="within")  # use actual variable names

# Perform Breusch-Pagan Test
bp_test <- bptest(fe_model)
print(bp_test)

# Interpret the Results
# If the p-value is small (e.g., p<0.05p<0.05): Reject the null hypothesis (H0: Homoscedasticity), indicating evidence of heteroscedasticity.
# If the p-value is large (e.g., p>0.05p>0.05): Fail to reject the null hypothesis (H0: Homoscedasticity), suggesting that the assumption of constant variance (homoscedasticity) is reasonable.
